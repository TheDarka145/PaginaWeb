CREATE TABLE Categoria (
    idCategoria INT PRIMARY KEY,
    NombreCategoria VARCHAR(50)
);

CREATE TABLE Planta (
    idPlanta CHAR(4) PRIMARY KEY,
    nombre VARCHAR(50),
    imagen VARCHAR(100),
    descripcion VARCHAR(100),
    categoria_id INT,
    precio INT,
    stock INT,
    FOREIGN KEY (categoria_id) REFERENCES Categoria(idCategoria) ON DELETE CASCADE
);

CREATE TABLE Boleta (
    id_boleta INT AUTO_INCREMENT PRIMARY KEY,
    total BIGINT,
    fechaCompra DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE detalle_boleta (
    id_detalle_boleta INT AUTO_INCREMENT PRIMARY KEY,
    id_boleta INT,
    id_producto CHAR(4),
    cantidad INT,
    subtotal BIGINT,
    FOREIGN KEY (id_boleta) REFERENCES Boleta(id_boleta) ON DELETE CASCADE,
    FOREIGN KEY (id_producto) REFERENCES Planta(idPlanta) ON DELETE CASCADE
);
